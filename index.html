<script>

const fmt = (d)=> d ? new Date(d.getTime()-d.getTimezoneOffset()*60000).toISOString().slice(0,10) : "—";

const parseDate = (s)=> { if(!s) return null; const d = new Date(s+"T12:00:00"); return isNaN(d)?null:d; };

const addDays = (d, n)=> { const r = new Date(d); r.setDate(r.getDate()+n); return r; };

const diffDays = (a,b)=> Math.round((a - b)/86400000);


function fromFUR(furStr, cycleLen){

  const fur = parseDate(furStr); if(!fur) return {error:"Introduce una FUR válida."};

  const adj = (Number(cycleLen)||28) - 28;

  const edd = addDays(fur, 280+adj);

  const today=new Date(); const gaDays=diffDays(today,fur);

  const w=Math.floor(gaDays/7), d=gaDays%7;

  return {edd,w,d};

}

function fromUS(scanStr,w,d){

  const scan=parseDate(scanStr); if(!scan) return {error:"Fecha eco inválida."};

  const ga=(+w||0)*7+(+d||0); if(ga<=0) return {error:"Semanas/días inválidos."};

  const edd=addDays(scan,280-ga); const today=new Date(); const lmp=addDays(scan,-ga);

  const g=diffDays(today,lmp); return {edd,w:Math.floor(g/7),d:g%7};

}


function showOut(el,data){

  if(data.error){el.style.display="block";el.innerHTML=data.error;return;}

  el.style.display="block"; el.innerHTML=`<b>FPP:</b> ${fmt(data.edd)}<br><b>Hoy:</b> ${data.w} sem + ${data.d} d`;

  document.getElementById('kpi').style.display='grid';

  document.getElementById('kpi_edd').textContent=fmt(data.edd);

  document.getElementById('kpi_ga').textContent=`${data.w}+${data.d}`;

}


document.getElementById('calcFUR').onclick=()=>{

  showOut(document.getElementById('outA'),fromFUR(fur.value,cycle.value));

  document.getElementById('outB').style.display='none';

}

document.getElementById('calcUS').onclick=()=>{

  showOut(document.getElementById('outB'),fromUS(scan.value,ga_w.value,ga_d.value));

  document.getElementById('outA').style.display='none';

}

document.getElementById('copy').onclick=()=>{

  const k=document.getElementById('kpi_edd').textContent;

  if(k!=="—") navigator.clipboard.writeText("Mi FPP: "+k);

}

document.getElementById('share').onclick=()=>{

  const url=location.href; navigator.clipboard.writeText(url); alert("Enlace copiado");

}
